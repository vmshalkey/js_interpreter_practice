Be the Interpreter

Below are the original and interpreted codes for some javascript functions to show the effects of variable hoisting and scoping.

Open Inspect Element and look at the Console to see the outputs for each of the functions.

Example

ORIGINAL CODE

	console.log(hello);
	var hello = "interesting";
	function example() {
		var hello = "hi!"
		console.log(hello)
	}
	example();
	console.log(hello)
INTERPRETED CODE

	var hello;
	console.log(hello);			// print: undefined
	hello = "interesting";
	function example() {
		var hello;
		hello = "hi";
		console.log(hello)		// print: "hi"
	}
	example();
	console.log(hello);			// print: "interesting"

Problem 1

ORIGINAL CODE

	console.log(first_variable);
	var first_variable = "Yipee I was first!";
	function firstFunc() {
		first_variable = "Not anymore!!!";
		console.log(first_variable)
	}
	console.log(first_variable);
INTERPRETED CODE

	var first_variable
	console.log(first_variable);			// print: undefined
	first_variable = "Yipee I was first!";
	function firstFunc() {
		first_variable = "Not anymore!!!";	// does not print (function not invoked)
		console.log(first_variable)
	}
	console.log(first_variable);			// print: "Yipee I was first!"

Problem 2

ORIGINAL CODE

	var food = "Chicken";
	function eat() {
		food = "half-chicken";
		console.log(food);
		var food = "gone";
	}
	eat();
	console.log(food);
INTERPRETED CODE

	var food;
	food = "Chicken";
	function eat() {
		var food = "gone";
		food = "half-chicken";
		console.log(food);		// print: "half-chicken"
	}
	eat();
	console.log(food);			// print: "Chicken"

Problem 3

ORIGINAL CODE

	var new_var = "NEW!";
	function lastFunc() {
		new_var = "old";
	}
	console.log(new_var);
INTERPRETED CODE

	var new_var;
	var new_var = "NEW!";
	function lastFunc() {
		new_var = "old";
	}
	console.log(new_var);		// print: "NEW!"
